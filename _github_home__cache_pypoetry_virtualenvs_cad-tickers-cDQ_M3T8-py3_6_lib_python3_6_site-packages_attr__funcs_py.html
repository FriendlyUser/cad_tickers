265def evolve(inst, **changes): 

266 """ 

267 Create a new instance, based on *inst* with *changes* applied. 

268 

269 :param inst: Instance of a class with ``attrs`` attributes. 

270 :param changes: Keyword changes in the new copy. 

271 

272 :return: A copy of inst with *changes* incorporated. 

273 

274 :raise TypeError: If *attr_name* couldn't be found in the class 

275 ``__init__``. 

276 :raise attr.exceptions.NotAnAttrsClassError: If *cls* is not an ``attrs`` 

277 class. 

278 

279 .. versionadded:: 17.1.0 

280 """ 

281 cls = inst.__class__ 

282 attrs = fields(cls) 

283 for a in attrs: 

284 if not a.init: 

285 continue 

286 attr_name = a.name # To deal with private attributes. 

287 init_name = attr_name if attr_name[0] != "_" else attr_name[1:] 

288 if init_name not in changes: 

289 changes[init_name] = getattr(inst, attr_name) 

290 return cls(**changes)
