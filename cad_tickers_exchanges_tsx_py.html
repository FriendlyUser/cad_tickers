1import requests 

2def get_mig_report(filename, exchange="TSX") -> str: 

3 """Gets excel spreadsheet from api.tsx using requests 

4 Input 

5 filename: Name of the file to be saved 

6 exchanges: TSX, TSXV 

7 Output: 

8 filePath returns path to file 

9 

10 See ://stackoverflow.com/questions/13567507/passing-csrftoken-with-python-requests 

11 """ 

12 tsx_url = 'https://api.tmxmoney.com/en/migreport/search' 

13 client = requests.session() 

14 # Retrieve the CSRF token first 

15 client.get(tsx_url) # sets cookie 

16 if 'csrftoken' in client.cookies: 

17 # Django 1.6 and up 

18 csrftoken = client.cookies['csrftoken'] 

19 else: 

20 # older versions 

21 csrftoken = client.cookies['csrf'] 

22 login_data = dict(report_type='excel', csrfmiddlewaretoken=csrftoken) 

23 if exchange in ["TSX", "TSXV"]: 

24 login_data["exchanges"] = exchange 

25 # To get all exchanges, leave blank 

26 # login_data = dict(report_type='excel', csrfmiddlewaretoken=client.cookies['csrftoken']) 

27 # Options are TSX, TSXV and nothing for all 

28 r = client.post("https://api.tmxmoney.com/en/migreport/search", data=login_data, headers=dict(Referer=tsx_url)) 

29 resp_headers = r.headers 

30 if "text/html" in resp_headers["Content-Type"]: 

31 return None 

32 elif resp_headers["Content-Type"] == "application/ms-excel": 

33 print(f"Downloaded data from {tsx_url} for {exchange} to {filename}") 

34 resp_data = r.content 

35 with open(filename, 'wb') as s: 

36 s.write(resp_data) 

37 return filename 

38 else: 

39 return None
